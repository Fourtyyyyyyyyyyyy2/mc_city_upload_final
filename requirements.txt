1# mc_city — runtime dependencies
# Tested on: Python 3.9.13 + Windows 11 (requires Python >= 3.9)
# Install:   python -m pip install -r requirements.txt
#
# Note: tkinter is part of the Python standard library (used by scan_gui) and
#       ships with the official installer, so it is not listed here.
#       On the Minecraft side you must also install the GDMC HTTP Interface mod
#       (Fabric/Forge) and have a world open.
#
# Verified: the packages below are the complete set of third-party imports in
#           this project — nothing is missing.

tqdm>=4.67,<5          # generation progress bar (main, scanner)
gdpc>=8.1,<91          # GDMC HTTP client: buildArea / players / version / runCommand
